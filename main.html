#include <stdio.h>
#include <time.h>
#define ROW 10
#define COL 10

int gen_rnd_nm(){             //create random number 1,2,3,4
int number;

  int r = ( rand() % 4 );
  return r;
}

char letter(){                        // Geef value van letter terug, elke keer als dit aangeroepen wordt verhoogd de letter.
static unsigned int call_count = 0;
call_count++;
char letter = 'B';
letter = 'B' - 1 + call_count;
return letter;
}


int main()
{
    int i,j,q,coin;
    int a=0;
    int b=0;
    int amountOfRuns = 25;       // aantal letter dat je gaat printen, 26-1
    srand(time(NULL));           //random seed

    char grid[ROW][COL];       //maak grid

    for (i = 0; i < ROW; i++) {
        for (j = 0; j < COL; j++) {

            grid[i][j] = '.';           //vul grid met .
            grid[0][0] = 'A';
        }
    }

          for (q = 0; q < amountOfRuns; q++){
            coin = gen_rnd_nm();               //get random number 0,1,2,3
            switch(coin){                      //kies een richting
            case 0  :
                a++;
                if (grid[a][b] == '.'&&(a<=9)){     //check of die plek leeg is en of het binnen het 10x10-grid valt.

                grid[a][b] = letter();               //assign letter aan die plek.
                }
                else{                                // als de plek niet valid is, run opnieuw, restore waarde
                    amountOfRuns++;
                    a--;
                }
                break;
            case 1 :
                a--;
                if (grid[a][b] == '.'&&(a>=0)){

                grid[a][b] = letter();
                }
                else{
                    amountOfRuns++;
                    a++;
                }
                break;
            case 2 :
                b++;

                grid[a][b] = letter();
                }
                else{
                    amountOfRuns++;
                    b--;
                }
                break;
            case 3 :
                b--;
                if (grid[a][b] == '.'&&(b>=0)){

                grid[a][b] = letter();
                }
                else{
                    amountOfRuns++;
                    b++;
                }
                break;
            }

if (amountOfRuns > 400){                       // als er na 400x proberen geen weg gevonden is: terminate

    for (i = 0; i < ROW; i++) {             //print grid
